```typescript
// app/api/checkout/razorpay/route.ts
export async function POST(req: Request) {
  const { userId } = await auth()
  const { planId } = await req.json()
  
  const org = await db.organization.findFirst({
    where: { members: { some: { userId } } }
  })
  
  const plan = PLANS[planId]
  
  // Create Razorpay order
  const order = await razorpay.orders.create({
    amount: plan.amount * 100, // In paise
    currency: 'INR',
    receipt: `org-${org.id}`,
    notes: {
      organizationId: org.id,
      userId,
      planId,
    }
  })
  
  return Response.json({
    orderId: order.id,
    amount: order.amount,
    currency: order.currency,
  })
}

// 4. Create webhook receiver
// app/api/webhooks/razorpay/route.ts
export async function POST(req: Request) {
  const signature = req.headers.get('x-razorpay-signature')
  const body = await req.text()
  
  // Verify webhook
  const verified = razorpay.utils.verifyWebhookSignature(
    body,
    signature!,
    process.env.RAZORPAY_WEBHOOK_SECRET!
  )
  
  if (!verified) {
    return new Response('Unauthorized', { status: 401 })
  }
  
  const event = JSON.parse(body)
  
  switch (event.event) {
    case 'payment.authorized':
      await handlePaymentAuthorized(event.payload)
      break
    case 'subscription.created':
      await handleSubscriptionCreated(event.payload)
      break
    case 'subscription.cancelled':
      await handleSubscriptionCancelled(event.payload)
      break
  }
  
  return Response.json({ success: true })
}
```

```typescript
// src/lib/payment/get-processor.ts
export function getPaymentProcessor(
  organizationRegion: string,
  userLocation: string
): 'razorpay' | 'stripe' {
  // Primary: User's location
  if (userLocation === 'IN') return 'razorpay'
  if (userLocation === 'US') return 'stripe'
  if (userLocation === 'EU') return 'stripe'
  
  // Fallback: Organization region
  if (organizationRegion === 'IN') return 'razorpay'
  
  // Default
  return 'stripe'
}

// app/checkout/page.tsx - Show correct processor
export default function CheckoutPage() {
  const processor = getPaymentProcessor(org.region, userLocation)
  
  return (
    <div>
      {processor === 'razorpay' && <RazorpayCheckout />}
      {processor === 'stripe' && <StripeCheckout />}
    </div>
  )
}
```

```typescript
// Redis-backed caching
export async function getCachedGoals(orgId: string) {
  const cached = await redis.get(`goals:${orgId}`)
  if (cached) return JSON.parse(cached)
  
  const goals = await db.goal.findMany({
    where: { organizationId: orgId }
  })
  
  await redis.setex(
    `goals:${orgId}`,
    CACHE_TTL.GOALS_LIST,
    JSON.stringify(goals)
  )
  
  return goals
}
```

```typescript
export const withTierRateLimit = (handler: Handler) => {
  return async (req: Request) => {
    const org = await getOrgFromRequest(req)
```
